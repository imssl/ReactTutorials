/D02DivReact.html

Create element with React.
Render element in an html div.

let divMadeWithReact =React.createElement('div', { style: { color: "red" } },'some text in div')
let domContainer = document.querySelector('#mainDiv');
ReactDOM.render(divMadeWithReact, domContainer);
